#include <config.h>

extern unsigned kernel_end;

void i386_mem_size()
{
        unsigned *tp, tmp, memsize;

        memsize = (unsigned)&kernel_end;

        do {
                tp  = (unsigned*)memsize;
                tmp = *tp;
                *tp = 0x55AA55AA;
                if(*tp != 0x55AA55AA) {
                        break;
                } else {
                        *tp = 0xAA55AA55;
                        if(*tp != 0xAA55AA55) {
                                break;
                        }
                }
                *tp = tmp;
                memsize+= 0x1000;
        } while(memsize < 0xFFFFFFFF);

        arch_memsize = memsize;
}
