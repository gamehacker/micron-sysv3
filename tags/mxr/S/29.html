/* debug functions */
#define PANIC(cond, str...) \
                if(cond) {\
                        kprintf("%CPANIC: %s: ", 0x0C, #cond); \
                        kprintf(str); \
                        kprintf("\n%C", 0x0F); \
                        while(1); \
                }
#define DEBUG(data)      kprintf("%CDEBUG:%s= %d\n%C",0x0C,#data,data,0x0F);
